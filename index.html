ระบบจะส่งผลการทำกิจกรรมไปยังชีท FingerMathData ผ่าน Web App ที่กำหนดไว้
เปิด “ส่งข้อมูลอัตโนมัติ” เพื่อบันทึกทันทีหลังจบกิจกรรม หรือกด “ส่งข้อมูลทั้งหมดตอนนี้” เพื่อส่งข้อมูลเดิมในเครื่อง 
function doGet() {
  return ContentService.createTextOutput(JSON.stringify({ok:true,message:'Finger Math API พร้อมใช้งาน'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents || '{}');
    var ss = SpreadsheetApp.openById('1XU_LRBLYwST83NJSEJKWYLeRxzz35N404V21QUxeiWc');
    var sheetName = data.sheetName || 'FingerMathData';
    var sheet = ss.getSheetByName(sheetName) || ss.insertSheet(sheetName);
    var rows = data.rows || [];
    if (sheet.getLastRow() === 0 && data.headers) sheet.appendRow(data.headers);
    if (rows.length) sheet.getRange(sheet.getLastRow()+1,1,rows.length,rows[0].length).setValues(rows);
    return ContentService.createTextOutput(JSON.stringify({ok:true,count:rows.length,message:'บันทึกข้อมูลสำเร็จ'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:String(err)}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
